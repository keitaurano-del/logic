2. useOnboardingState Hook

// hooks/useOnboardingState.ts
import { useState, useCallback } from 'react';
import type { OnboardingState, Goal, BillingPlan, OnboardingScreen } from '../types/onboarding';

const STORAGE_KEY = 'logic_onboarding_v1';

const defaultState: OnboardingState = {
  hasSeenWelcome: false,
  selectedGoal: null,
  hasSeenPlacementIntro: false,
  hasCompletedPlacement: false,
  placementScore: null,
  hasSeenAiPreview: false,
  hasSeenPaywall: false,
  subscriptionStatus: 'none',
  trialStartedAt: null,
  selectedBillingPlan: null,
};

export function useOnboardingState() {
  const [state, setState] = useState<OnboardingState>(() => {
    try {
      const saved = localStorage.getItem(STORAGE_KEY);
      return saved ? { ...defaultState, ...JSON.parse(saved) } : defaultState;
    } catch {
      return defaultState;
    }
  });

  const update = useCallback((patch: Partial<OnboardingState>) => {
    setState(prev => {
      const next = { ...prev, ...patch };
      try { localStorage.setItem(STORAGE_KEY, JSON.stringify(next)); } catch { /* quota exceeded */ }
      return next;
    });
  }, []);

  const reset = useCallback(() => {
    try { localStorage.removeItem(STORAGE_KEY); } catch { /* ignore */ }
    setState(defaultState);
  }, []);

  /** アプリ起動時にどの画面に戻るか決定する */
  const getResumeScreen = useCallback((): OnboardingScreen => {
    if (state.subscriptionStatus === 'active' || state.subscriptionStatus === 'trial') return 'home';
    if (!state.hasSeenWelcome) return 'welcome';
    if (!state.hasCompletedPlacement) {
      return state.hasSeenPlacementIntro ? 'placement_intro' : 'goal';
    }
    if (!state.hasSeenAiPreview) return 'ai_preview';
    if (!state.hasSeenPaywall && state.subscriptionStatus === 'none') return 'paywall';
    return 'home';
  }, [state]);

  return { state, update, reset, getResumeScreen };
}

3. OnboardingNavigator

// components/OnboardingNavigator.tsx
import React, { useState } from 'react';
import { useOnboardingState } from '../hooks/useOnboardingState';
import { trackEvent } from '../lib/analytics';
import type { Goal, BillingPlan, OnboardingScreen } from '../types/onboarding';

const RECOMMENDED_PATH = [
  { id: 'logic-tree', title: 'ロジックツリー', description: '問いを構造化する土台を作る' },
  { id: 'fermi', title: 'フェルミ推定', description: '分解 → 仮説 → 数値感覚を実践で鍛える' },
  { id: 'conclusion-first', title: '結論ファースト', description: '面接や報告で伝わる話し方を身につける' },
];

export function OnboardingNavigator({ onComplete }: { onComplete: () => void }) {
  const { state, update, getResumeScreen } = useOnboardingState();
  const [screen, setScreen] = useState<OnboardingScreen>(getResumeScreen());
  const [selectedPlan, setSelectedPlan] = useState<BillingPlan>('annual');
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const go = (s: OnboardingScreen) => setScreen(s);

  // ── Handlers ──────────────────────────────────────
  const onWelcomeStart = () => {
    update({ hasSeenWelcome: true });
    trackEvent('onboarding_welcome_view', {});
    go('goal');
  };

  const onGoalSelect = (goal: Goal) => {
    update({ selectedGoal: goal });
    trackEvent('onboarding_goal_selected', { selected_goal: goal });
  };

  const onGoalNext = () => {
    trackEvent('onboarding_goal_view', { selected_goal: state.selectedGoal ?? 'none' });
    go('placement_intro');
  };

  const onPlacementStart = () => {
    update({ hasSeenPlacementIntro: true });
    trackEvent('onboarding_placement_started', { selected_goal: state.selectedGoal ?? 'none' });
    // TODO: 実際のテスト画面へ遷移。今はシミュレーション:
    setTimeout(() => {
      update({ hasCompletedPlacement: true, placementScore: 48 });
      trackEvent('onboarding_placement_completed', { score_band: 'mid' });
      go('placement_result');
    }, 0);
  };

  const onResultTryAI = () => {
    trackEvent('onboarding_result_view', { score_band: 'mid' });
    go('ai_preview');
  };

  const onAiPreviewTrial = () => {
    update({ hasSeenAiPreview: true });
    trackEvent('onboarding_ai_preview_view', {});
    go('paywall');
  };

  const onStartTrial = async () => {
    setIsLoading(true);
    setError(null);
    trackEvent('trial_started', { billing_plan: selectedPlan });
    try {
      // TODO: Stripe / Google Play / App Store API call
      // await purchaseSubscription(selectedPlan);
      update({
        subscriptionStatus: 'trial',
        trialStartedAt: new Date().toISOString(),
        selectedBillingPlan: selectedPlan,
        hasSeenPaywall: true,
      });
      onComplete();
    } catch (e) {
      setError(e instanceof Error ? e.message : '購入処理に失敗しました。もう一度お試しください。');
    } finally {
      setIsLoading(false);
    }
  };

  const onContinueFree = () => {
    update({ hasSeenPaywall: true });
    trackEvent('free_continue_selected', { source_screen: screen });
    onComplete();
  };

  // ── Render ──────────────────────────────────────
  switch (screen) {
    case 'welcome':
      return <WelcomeScreen onStart={onWelcomeStart} onSkip={() => go('goal')} />;
    case 'goal':
      return <GoalSelectionScreen selectedGoal={state.selectedGoal} onSelect={onGoalSelect} onNext={onGoalNext} onSkip={() => go('placement_intro')} />;
    case 'placement_intro':
      return <PlacementIntroScreen onStart={onPlacementStart} onTryOne={onPlacementStart} />;
    case 'placement_result':
      return <PlacementResultScreen score={state.placementScore ?? 48} summary="分解の筋は良いですが、前提の置き方に伸びしろがあります" recommendedPath={RECOMMENDED_PATH} onTryAI={onResultTryAI} onGoHome={onContinueFree} />;
    case 'ai_preview':
      return <AIPreviewScreen onStartTrial={onAiPreviewTrial} onContinueFree={onContinueFree} />;
    case 'paywall':
      return <PaywallScreen selectedPlan={selectedPlan} onSelectPlan={setSelectedPlan} onStartTrial={onStartTrial} onContinueFree={onContinueFree} isLoading={isLoading} error={error} />;
    default:
      return null;
  }
}

interface OnboardingScaffoldProps {
  currentStep: number;   // 1-6
  totalSteps?: number;   // default 6
  onBack?: () => void;
  onSkip?: () => void;
  children: React.ReactNode;
  stickyFooter?: React.ReactNode;
}

export function OnboardingScaffold({
  currentStep, totalSteps = 6, onBack, onSkip, children, stickyFooter
}: OnboardingScaffoldProps) {
  return (
    <div style={{ display: 'flex', flexDirection: 'column', minHeight: '100dvh', background: 'var(--color-white)' }}>
      {/* Header */}
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', padding: '16px 20px' }}>
        {onBack
          ? <button onClick={onBack} style={{ background: 'none', border: 'none', cursor: 'pointer', padding: '8px' }}>
              <ChevronLeftIcon />
            </button>
          : <div style={{ width: 40 }} />
        }
        <ProgressDots current={currentStep} total={totalSteps} />
        {onSkip
          ? <button onClick={onSkip} style={{ background: 'none', border: 'none', cursor: 'pointer', color: 'var(--color-text-muted)', fontSize: 14 }}>
              スキップ
            </button>
          : <div style={{ width: 40 }} />
        }
      </div>
      {/* Scrollable content */}
      <div style={{ flex: 1, overflowY: 'auto', padding: '8px 20px 24px' }}>
        {children}
      </div>
      {/* Sticky footer */}
      {stickyFooter && (
        <div style={{ padding: '12px 20px 32px', borderTop: '1px solid var(--color-gray-border)', background: 'var(--color-white)' }}>
          {stickyFooter}
        </div>
      )}
    </div>
  );
}

// lib/analytics.ts
export type OnboardingEventName =
  | 'onboarding_welcome_view'
  | 'onboarding_goal_view'
  | 'onboarding_goal_selected'
  | 'onboarding_placement_intro_view'
  | 'onboarding_placement_started'
  | 'onboarding_placement_completed'
  | 'onboarding_result_view'
  | 'onboarding_ai_preview_view'
  | 'onboarding_paywall_view'
  | 'paywall_annual_selected'
  | 'paywall_monthly_selected'
  | 'trial_started'
  | 'paywall_dismissed'
  | 'free_continue_selected'
  | 'first_lesson_started'
  | 'first_ai_feedback_used'
  | 'first_fermi_started'
  | 'first_roleplay_started';

export interface EventParams {
  selected_goal?: string;
  score_band?: 'low' | 'mid' | 'high';  // <40 / 40-60 / >60
  billing_plan?: 'annual' | 'monthly';
  source_screen?: string;
  subscription_status?: string;
}

export function trackEvent(name: OnboardingEventName, params: EventParams): void {
  // Firebase Analytics via Capacitor plugin
  try {
    if (typeof window !== 'undefined' && (window as Record<string,unknown>).FirebaseAnalytics) {
      ((window as Record<string,unknown>).FirebaseAnalytics as { logEvent: (p: unknown) => void })
        .logEvent({ name, params });
    }
  } catch { /* ignore */ }
  if (import.meta.env.DEV) console.log('[Analytics]', name, params);
}
